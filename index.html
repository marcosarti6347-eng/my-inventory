// [최종 완전판] 브라우저 순번 기억 버그까지 완전히 박멸한 복원 로직 function importDatabase(input) { if (!input.files || !input.files[0]) { focusInput(); return; } const file = input.files[0]; const reader = new FileReader(); reader.onload = (e) => { try { const importedData = JSON.parse(e.target.result); if (!Array.isArray(importedData)) { alert("올바른 장부 백업 파일이 아닙니다."); input.value = ""; focusInput(); return; } if (confirm(`⚠️ 최종 경고: 현재 핸드폰에 남아있는 모든 장부 기록을 흔적도 없이 삭제하고, 백업 파일의 [${importedData.length}개] 데이터로 완벽히 덮어씁니다. 진행하시겠습니까?`)) { // 1단계: 기존 연결을 닫고 데이터베이스 자체를 완전히 삭제하여 일련번호 카운터까지 리셋합니다. db.close(); const deleteRequest = indexedDB.deleteDatabase("BackupEnabledMatchDB"); deleteRequest.onsuccess = () => { // 2단계: DB가 완전히 삭제되면, 처음부터 다시 깨끗하게 방을 만듭니다. const reopenRequest = indexedDB.open("BackupEnabledMatchDB", 1); reopenRequest.onupgradeneeded = (event) => { const newDb = event.target.result; const store = newDb.createObjectStore("checks", { keyPath: "id", autoIncrement: true }); store.createIndex("code", "code", { unique: false }); }; reopenRequest.onsuccess = (event) => { db = event.target.result; // 전역 db 객체 갱신 // 3단계: 뻥 뚫린 새 방에 백업 데이터를 1번부터 차곡차곡 집어넣습니다. const tx = db.transaction("checks", "readwrite"); const store = tx.objectStore("checks"); importedData.forEach((item, index) => { // id를 1부터 강제로 지정하여 기존 좀비 데이터와 결합되는 것을 원천 차단합니다. store.add({ id: index + 1, code: item.code, photo: item.photo, date: item.date }); }); tx.oncomplete = () => { alert("📱 핸드폰 내부 장부를 완벽히 초기화한 후, 백업 데이터만 깔끔하게 복원했습니다!"); input.value = ""; loadHistory(); // 화면 UI 리셋 document.getElementById('productCode').value = ""; searchCode(""); focusInput(); }; tx.onerror = () => { alert("데이터를 새로 쓰는 중에 오류가 발생했습니다."); input.value = ""; focusInput(); }; }; reopenRequest.onerror = () => { alert("장부 재구축에 실패했습니다. 앱을 새로고침 해주세요."); }; }; deleteRequest.onerror = () => { alert("기존 장부 데이터를 파괴하는 데 실패했습니다."); input.value = ""; focusInput(); }; } else { input.value = ""; focusInput(); } } catch (err) { alert("파일을 읽는 도중 오류가 발생했습니다. 올바른 백업 파일인지 확인해주세요."); input.value = ""; focusInput(); } }; reader.readAsText(file); }
